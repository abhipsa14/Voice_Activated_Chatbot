# ── Voice Activity Detection ──────────────────────────────────
PyAudio                 # Microphone access (needs portaudio)
webrtcvad               # Google WebRTC Voice Activity Detection

# ── Speech-to-Text ────────────────────────────────────────────
SpeechRecognition       # Fallback STT (Google Speech API)
# openai-whisper        # ⚠️ DESKTOP ONLY — do NOT install on Pi

# ── LLM ───────────────────────────────────────────────────────
requests                # HTTP client for Ollama API

# ── Caching ───────────────────────────────────────────────────
cachetools              # In-memory TTL cache (L1)
redis                   # Redis client for L3 cache
joblib                  # Serialisation for TF-IDF cache

# ── Text-to-Speech ────────────────────────────────────────────
# piper-tts             # Piper neural TTS (install separately on Pi)
edge-tts                # Microsoft Edge neural TTS (fallback, needs internet)
pygame                  # Audio playback for Edge TTS / cached audio
pyttsx3                 # Offline fallback TTS (espeak / SAPI5)
